    def Binary.type_bytes(type,type_len)
      optional_int = ""

      if(type_len < 15) then
        type += sprintf("%x",type_len)
      else
        type += "f"
        optional_int = Binary.int_bytes(type_len)
      end

      return [type].pack("H*") + optional_int
    end
